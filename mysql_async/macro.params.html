macro_rules! params {
    () => { ... };
    (@to_pair $map:expr, $name:expr => $value:expr) => { ... };
    (@to_pair $map:expr, $name:ident) => { ... };
    (@expand $map:expr;) => { ... };
    (@expand $map:expr; $name:expr => $value:expr, $($tail:tt)*) => { ... };
    (@expand $map:expr; $name:expr => $value:expr $(, $tail:tt)*) => { ... };
    (@expand $map:expr; $name:ident, $($tail:tt)*) => { ... };
    (@expand $map:expr; $name:ident $(, $tail:tt)*) => { ... };
    ($i:ident, $($tail:tt)*) => { ... };
    ($i:expr => $($tail:tt)*) => { ... };
    ($i:ident) => { ... };
}
